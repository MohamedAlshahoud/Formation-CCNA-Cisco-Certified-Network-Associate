!---------------------------------------------
! Configuration de base et Securite
!---------------------------------------------
en
conf t
hostname R2
no ip domain-lookup
ip domain-name labo-connecting.com
enable secret class
security password min-length 5
service password-encryption
username admin secret admin123
banner motd "EQUIPEMENT ISP - ACCES RESERVE"

!---------------------------------------------
! ACL pour SSH (Creation manquante)
!---------------------------------------------
ip access-list standard ACL-ADMIN-ONLY
 permit 172.16.30.0 0.0.0.255
 permit any 
exit

!---------------------------------------------
! Configuration des Lignes (Console et SSH)
!---------------------------------------------
line console 0
 password cisco
 login
exit

!---------------------------------------------
! Generation de la cle pour SSH
!---------------------------------------------
crypto key generate rsa


line vty 0 15
 login local
 transport input ssh
 access-class ACL-ADMIN-ONLY in 
exit

!---------------------------------------------
! Interface WAN (Vers R1)
!---------------------------------------------
interface f0/0
 description LINK_TO_CLIENT_R1
 ip address 209.165.200.226 255.255.255.224
 no shutdown
exit


!---------------------------------------------
! Route de retour vers le Pool NAT de R1
!---------------------------------------------
ip route 209.165.200.224 255.255.255.224 209.165.200.225

end
