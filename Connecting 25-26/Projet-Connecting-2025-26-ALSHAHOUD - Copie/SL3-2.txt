!---------------------------------------------
! Configuration de base et Securite
!---------------------------------------------
en
conf t
hostname SL3-2
no ip domain-lookup
ip domain-name labo-connecting.com
enable secret class
security password min-length 5
service password-encryption
username admin secret admin123
banner motd "Acces reserve aux administrateurs - SL3-2"

!---------------------------------------------
! ACL de restriction Management (VTY)
!---------------------------------------------
ip access-list standard ACL-ADMIN-ONLY
 permit 172.16.30.0 0.0.0.255
exit

!---------------------------------------------
! Configuration des Lignes (Console et SSH)
!---------------------------------------------
line console 0
 password cisco
 login
exit

!---------------------------------------------
! Generation de la cle pour SSH
!---------------------------------------------
crypto key generate rsa


line vty 0 15
 login local
 transport input ssh
 access-class ACL-ADMIN-ONLY in 
exit

!---------------------------------------------
! Configuration DHCP (POUR PC2 et PC3)
!---------------------------------------------

ip dhcp excluded-address 172.16.20.1 172.16.20.10
ip dhcp excluded-address 172.16.30.1 172.16.30.20
ip dhcp excluded-address 172.16.30.100

!---------------------------------------------
! Pool pour PC2 (STAFF)
!---------------------------------------------
ip dhcp pool POOL_STAFF
 network 172.16.20.0 255.255.255.0
 default-router 172.16.20.1
 dns-server 172.16.30.100


!---------------------------------------------
! Pool pour PC3 (MGNT)
!---------------------------------------------
ip dhcp pool POOL_MGNT
 network 172.16.30.0 255.255.255.0
 default-router 172.16.30.1
 dns-server 172.16.30.100
exit

!---------------------------------------------
! Configuration des VLANs (Tous pour OSPF)
!---------------------------------------------
vlan 10
 name RD
vlan 20
 name STAFF
vlan 30
 name MGNT
vlan 99
 name NATIF
exit

!---------------------------------------------
! Routage Inter-VLAN (SVI)
!---------------------------------------------
ip routing

interface Vlan 20
 description GATEWAY_VLAN20
 ip address 172.16.20.1 255.255.255.0

 !---------------------------------------------
 ! On bloque ici le PC2 avant qu'il n'entre dans le routeur
 !---------------------------------------------
 ip access-group BLOQUER-STAFF-SERVEUR in

interface Vlan 30
 description GATEWAY_VLAN30
 ip address 172.16.30.1 255.255.255.0
exit

!---------------------------------------------
! ACL de restriction (PC2 vers Serveur0)
!---------------------------------------------
ip access-list extended BLOQUER-STAFF-SERVEUR
 deny ip 172.16.20.0 0.0.0.255 host 172.16.30.100
 permit ip any any
exit

!---------------------------------------------
! Liaison Trunk et Port Serveur
!---------------------------------------------
interface g1/0/2
 description LINK_TO_S2
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk native vlan 99

interface f1/0/1
 description SERVEUR_0
 switchport mode access
 switchport access vlan 30
 spanning-tree portfast
exit

!---------------------------------------------
! Routage Dynamique OSPF
!---------------------------------------------
router ospf 1
 router-id 3.3.3.2
 network 172.16.20.0 0.0.0.255 area 0
 network 172.16.30.0 0.0.0.255 area 0
exit

end
