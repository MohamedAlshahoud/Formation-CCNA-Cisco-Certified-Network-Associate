!---------------------------------------------
! Configuration de base et Securite
!---------------------------------------------
en
conf t
hostname SL3-1
no ip domain-lookup
ip domain-name labo-connecting.com
enable secret class
security password min-length 5
service password-encryption
username admin secret admin123
banner motd "Acces reserve aux administrateurs - SL3-1"

!---------------------------------------------
! ACL de restriction Management (VTY)
!---------------------------------------------
ip access-list standard ACL-ADMIN-ONLY
 permit 172.16.30.0 0.0.0.255
exit

!---------------------------------------------
! Configuration des Lignes (Console et SSH)
!---------------------------------------------
line console 0
 password cisco
 login
exit

!---------------------------------------------
! Generation de la cle pour SSH
!---------------------------------------------
crypto key generate rsa


line vty 0 15
 login local
 transport input ssh
 access-class ACL-ADMIN-ONLY in 
exit

!---------------------------------------------
! Configuration DHCP (POUR PC1 - VLAN 10)
!---------------------------------------------
ip dhcp excluded-address 172.16.10.1 172.16.10.10

ip dhcp pool POOL_RD
 network 172.16.10.0 255.255.255.0
 default-router 172.16.10.1
 dns-server 172.16.30.100
exit

!---------------------------------------------
! Configuration des VLANs
!---------------------------------------------
vlan 10
 name RD
vlan 20
 name STAFF
vlan 30
 name MGNT
vlan 99
 name NATIF
exit

!---------------------------------------------
! Routage Inter-VLAN (SVI)
!---------------------------------------------
ip routing

interface Vlan 10
 description GATEWAY_VLAN10
 ip address 172.16.10.1 255.255.255.0
 ip access-group FILTRAGE-PC1-RD in

interface Vlan 30
 description GATEWAY_VLAN30
 ip address 172.16.30.2 255.255.255.0
exit

!---------------------------------------------
! ACL de restriction (Securite PC1 vers Serveur)
!---------------------------------------------
ip access-list extended FILTRAGE-PC1-RD
 deny ip 172.16.10.0 0.0.0.255 host 172.16.30.100
 permit ip any any
exit

!---------------------------------------------
! Interfaces Trunk et Liaison R1
!---------------------------------------------
interface range g1/0/1 - 2
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk native vlan 99
exit

interface f0/1
 description LINK_TO_R1
 no switchport
 ip address 10.10.13.1 255.255.255.252
exit

!---------------------------------------------
! Routage Dynamique OSPF
!---------------------------------------------
router ospf 1
 router-id 3.3.3.1
 network 172.16.10.0 0.0.0.255 area 0
 network 172.16.30.0 0.0.0.255 area 0
 network 10.10.13.0 0.0.0.3 area 0
exit

end
