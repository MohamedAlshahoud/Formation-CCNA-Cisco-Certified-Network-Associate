!---------------------------------------------
! Configuration de base et Securite
!---------------------------------------------
en
conf t
hostname R1
no ip domain-lookup
ip domain-name labo-connecting.com
enable secret class
security password min-length 5
service password-encryption
username admin secret admin123
banner motd "Acces reserve aux administrateurs - R1"

!---------------------------------------------
! ACL de restriction Management (CREEE AVANT VTY)
!---------------------------------------------
ip access-list standard ACL-VTY-ADMIN
 permit 172.16.30.0 0.0.0.255
exit

!---------------------------------------------
! Configuration des Lignes (Console et SSH)
!---------------------------------------------
line console 0
 password cisco
 login
exit

!---------------------------------------------
! Generation de la cle pour SSH
!---------------------------------------------

crypto key generate rsa

line vty 0 15
 login local
 transport input ssh
 access-class ACL-VTY-ADMIN in 
exit

!---------------------------------------------
! Interfaces LAN et WAN
!---------------------------------------------
interface f0/0
 description LINK_TO_INTERNAL_CORE
 ip address 10.10.13.2 255.255.255.252
 ip nat inside
 no shutdown
exit

interface f2/0
 description LINK_TO_INTERNET_WAN
 ip address 209.165.200.225 255.255.255.224
 ip nat outside
 no shutdown
exit

!---------------------------------------------
! Configuration NAT (Overload / PAT)
!---------------------------------------------

access-list 1 permit 172.16.0.0 0.0.255.255
ip nat inside source list 1 interface f2/0 overload

!---------------------------------------------
! Routage Dynamique (OSPF) et Route Statique
!---------------------------------------------
router ospf 1
 router-id 1.1.1.1
 network 10.10.13.0 0.0.0.3 area 0
 
 !---------------------------------------------
 ! Diffuse la route par defaut aux autres routeurs (SL3-1/SL3-2)
 !---------------------------------------------
 default-information originate
exit

!---------------------------------------------
! Route par defaut vers le prochain saut (R2 ISP)
!---------------------------------------------
ip route 0.0.0.0 0.0.0.0 209.165.200.226

end
