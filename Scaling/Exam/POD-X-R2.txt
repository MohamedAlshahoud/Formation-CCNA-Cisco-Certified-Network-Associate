!---------------------------------------------
! Configuration
!---------------------------------------------
enable
configure terminal
hostname POD-X-R2
no ip domain-lookup
ip domain-name isp-public.com
enable secret class
security password min-length 5
service password-encryption

username admin secret admin123
banner motd "Acces reserve aux administrateurs - ISP - Exam Session 1"

!---------------------------------------------
! Configuration SSH
!---------------------------------------------
crypto key generate rsa modulus 1024

line vty 0 15
 login local
 transport input ssh
exit

!---------------------------------------------
! Interface Simulation Internet
!---------------------------------------------
! Cette IP (209.165.100.1) sert de cible de test pour le NAT
!---------------------------------------------
interface Loopback0
 description INTERNET_SIMULE
 ip address 209.165.100.1 255.255.255.255
exit

!---------------------------------------------
! Interface de Liaison (Vers POD-X-R1)
!---------------------------------------------
! Utilise la "deuxieme adresse" du pool 209.165.201.0/27
!---------------------------------------------
interface FastEthernet0/0
 description LIEN_VERS_POD-X-R1
 ip address 209.165.201.2 255.255.255.224
 ip access-group ACL-ISP-IN in
 no shutdown
exit

!---------------------------------------------
! ACL Anti-Spoofing (Mise a jour pour l'examen)
!---------------------------------------------
! Bloque les paquets entrants ayant des IPs privees source
!---------------------------------------------
ip access-list extended ACL-ISP-IN
 deny ip 10.0.0.0 0.255.255.255 any
 deny ip 172.16.0.0 0.15.255.255 any
 deny ip 192.168.0.0 0.0.255.255 any
 permit ip any any
exit

!---------------------------------------------
! Route de retour vers le pool public du NAT
!---------------------------------------------
! Necessaire pour que l'ISP sache ou renvoyer le trafic 
! destiné aux IPs publiques de POD-X-R1
!---------------------------------------------
ip route 209.165.201.0 255.255.255.224 209.165.201.1

end
write
