!---------------------------------------------
! Configuration de base et Securite (Examen R2)
!---------------------------------------------
enable
configure terminal
hostname R2
no ip domain-lookup
ip domain-name labo-NAT.com
enable secret class
security password min-length 5
service password-encryption

username admin secret admin123
banner motd "Acces reserve aux administrateurs - R2 - Exam Session 1"

!---------------------------------------------
! Configuration des Lignes (Console et SSH)
!---------------------------------------------
line console 0
 password cisco
 login
exit

! Generation de la cle RSA pour SSH (Module 1024)
crypto key generate rsa modulus 1024

line vty 0 15
 login local
 transport input ssh
 access-class ACL-ADMIN-ONLY in
exit

!---------------------------------------------
! Configuration des sous-interfaces (LAN 192.168.x.x)
!---------------------------------------------
interface g0/0/0
 description LIAISON_VERS_S2
 no shutdown
exit

! VLAN 10 - R&D
interface g0/0/0.10
 description VLAN_RD
 encapsulation dot1Q 10
 ip address 192.168.10.3 255.255.255.0
 ip access-group ACL-VLAN10 in
 ip helper-address 172.16.23.2
 standby 10 ip 192.168.10.1
 standby 10 priority 100
 standby 10 preempt
exit

! VLAN 20 - STAFF
interface g0/0/0.20
 description VLAN_STAFF
 encapsulation dot1Q 20
 ip address 192.168.20.3 255.255.255.0
 ip access-group ACL-VLAN20 in
 ip helper-address 172.16.23.2
 standby 20 ip 192.168.20.1
 standby 20 priority 100
 standby 20 preempt
exit

! VLAN 30 - MGNT
interface g0/0/0.30
 description VLAN_MGNT
 encapsulation dot1Q 30
 ip address 192.168.30.3 255.255.255.0
 ip access-group ACL-VLAN30 in
 standby 30 ip 192.168.30.1
 standby 30 priority 100
 standby 30 preempt
exit

! VLAN 40 - DIR
interface g0/0/0.40
 description VLAN_DIR
 encapsulation dot1Q 40
 ip address 192.168.40.3 255.255.255.0
 standby 40 ip 192.168.40.1
 standby 40 priority 100
 standby 40 preempt
exit

! VLAN 99 - NATIF
interface g0/0/0.99
 description VLAN_NATIF
 encapsulation dot1Q 99 native
exit

!---------------------------------------------
! Liaison WAN (Vers POD-X-R1)
!---------------------------------------------
interface g0/0/1
 description LINK_TO_POD-X-R1
 ip address 172.16.23.1 255.255.255.252
 no shutdown
exit

!---------------------------------------------
! Routage OSPF
!---------------------------------------------
router ospf 1
 router-id 2.2.2.2
 network 192.168.10.0 0.0.0.255 area 0
 network 192.168.20.0 0.0.0.255 area 0
 network 192.168.30.0 0.0.0.255 area 0
 network 192.168.40.0 0.0.0.255 area 0
 network 172.16.23.0 0.0.0.3 area 0
exit

!---------------------------------------------
! ACLs de Politique de Securite
!---------------------------------------------

! Restriction SSH au VLAN MGNT uniquement
ip access-list standard ACL-ADMIN-ONLY
 permit 192.168.30.0 0.0.0.255
exit

! ACL VLAN 10 : Acces Internet + Accès exclusif Loopback R3 (10.10.10.1)
ip access-list extended ACL-VLAN10
 permit ip 192.168.10.0 0.0.0.255 host 10.10.10.1 
 deny ip 192.168.10.0 0.0.0.255 192.168.20.0 0.0.0.255 
 deny ip 192.168.10.0 0.0.0.255 192.168.30.0 0.0.0.255 
 permit ip any any                                     
exit

! ACL VLAN 20 : Acces Internet
ip access-list extended ACL-VLAN20
 deny ip 192.168.20.0 0.0.0.255 host 10.10.10.1        
 deny ip 192.168.20.0 0.0.0.255 192.168.10.0 0.0.0.255 
 deny ip 192.168.20.0 0.0.0.255 192.168.30.0 0.0.0.255 
 permit ip any any                                     
exit

! ACL VLAN 30 : Acces Management
ip access-list extended ACL-VLAN30
 deny ip 192.168.30.0 0.0.0.255 192.168.10.0 0.0.0.255 
 deny ip 192.168.30.0 0.0.0.255 192.168.20.0 0.0.0.255 
 permit ip any any                                     
exit

end
write
