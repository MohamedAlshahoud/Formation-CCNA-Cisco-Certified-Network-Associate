!---------------------------------------------
! Configuration de base et Securite
!---------------------------------------------
enable
configure terminal
hostname POD-X-R2
no ip domain-lookup
ip domain-name isp-public.com
enable secret class
security password min-length 5
service password-encryption

username admin secret admin123
banner motd "Acces reserve aux administrateurs - isp-public.com"

!---------------------------------------------
! ACL de restriction Management
!---------------------------------------------
!---------------------------------------------
! Seul le VLAN MGNT (172.16.30.0) peut se connecter en SSH
!---------------------------------------------

ip access-list standard ACL-ADMIN-ONLY
 permit 172.16.30.0 0.0.0.255
exit

!---------------------------------------------
! Configuration des Lignes (Console et SSH)
!---------------------------------------------
line console 0
 password cisco
 login
exit


!---------------------------------------------
! Generation de la cle pour SSH
!---------------------------------------------

crypto key generate rsa


line vty 0 15
 login local
 transport input ssh
 access-class ACL-ADMIN-ONLY in
exit

!---------------------------------------------
! Interface Simulation Internet
!---------------------------------------------

interface Loopback0
 description INTERNET_SIMULE
 ip address 209.165.100.1 255.255.255.255

!---------------------------------------------
! Interface Liaison Client (Vers POD-X-R1)
!---------------------------------------------

interface GigabitEthernet0/0
 description LIEN_VERS_CLIENT
 ip address 209.165.200.226 255.255.255.224
 ip access-group ACL-ISP-IN in
 duplex auto
 speed auto
 no shutdown

!---------------------------------------------
! Routage Statique de Retour
!---------------------------------------------

ip route 209.165.200.224 255.255.255.224 209.165.200.225


!---------------------------------------------
! ACL Anti-Spoofing
!---------------------------------------------

ip access-list extended ACL-ISP-IN
 deny ip 10.0.0.0 0.255.255.255 any
 deny ip 172.16.0.0 0.15.255.255 any
 deny ip 192.168.0.0 0.0.255.255 any
 permit ip any any

end
