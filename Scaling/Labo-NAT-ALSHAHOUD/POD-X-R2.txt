!---------------------------------------------
! Configuration POD-X-R2 (ISP)
!---------------------------------------------
enable
configure terminal
hostname POD-X-R2
no ip domain-lookup
ip domain-name isp-public.com
enable secret class
security password min-length 5
service password-encryption

username admin secret admin123
banner motd "Acces reserve aux administrateurs - ISP"

!---------------------------------------------
! SSH 
!---------------------------------------------
crypto key generate rsa

line vty 0 15
 login local
 transport input ssh
exit

!---------------------------------------------
! Interface Simulation Internet 
!---------------------------------------------
! C'est cette IP que vous pinguerez pour tester le NAT
!---------------------------------------------
interface Loopback0
 description INTERNET_SIMULE
 ip address 209.165.100.1 255.255.255.255
exit

!---------------------------------------------
! Interface de Liaison (Vers POD-X-R1)
!---------------------------------------------
! Sur le routeur virtuel, verifiez si c'est f0/0 ou g0/0
!---------------------------------------------
interface FastEthernet0/0
 description LIEN_VERS_POD-X-R1
 ip address 209.165.200.226 255.255.255.224
 ip access-group ACL-ISP-IN in
 no shutdown
exit

!---------------------------------------------
! ACL Anti-Spoofing 
!---------------------------------------------
! Bloque les adresses privees qui ne devraient pas etre sur Internet
!---------------------------------------------
ip access-list extended ACL-ISP-IN
 deny ip 10.0.0.0 0.255.255.255 any
 deny ip 172.16.0.0 0.15.255.255 any
 deny ip 192.168.0.0 0.0.255.255 any
 permit ip any any
exit

!---------------------------------------------
! Route de retour vers votre Pool Public 
!---------------------------------------------
! Indispensable pour que POD-X-R2 sache renvoyer les paquets vers 
! POD-X-R1
!---------------------------------------------
ip route 209.165.200.224 255.255.255.224 209.165.200.225

end
wr
